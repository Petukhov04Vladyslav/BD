1. Аналіз універсального відношення:

    Головний ключ : Номер
    Обов'язкові поля для введення: Номер, Код, Новинка, Назва, Видавництво, Сторінки, Тираж, Тема, Категорія
    Значення за замовчуванням: Формат (000х000/00), Ціна (NULL), Дата (NULL).
    Область допустимих значень: Може визначатися для числових полів (Номер, Код, Тираж, Сторінки) та для перелічуваних типів (Новинка, Категорія, Видавництво, Тема).
    Індексація: Код.

2. a. Видалення попередньої версії бази даних

DROP DATABASE IF EXISTS lab_;

-- b. Створення бази даних
CREATE DATABASE lab;

-- Перехід до бази даних
USE lab;

-- c. Створення таблиці на основі універсального відношення
CREATE TABLE table_1 (
    № INT(11) NOT NULL,
    Code INT(11) PRIMARY KEY,
    New VARCHAR(3) NOT NULL,
    NAME TEXT NOT NULL,
    Price FLOAT DEFAULT NULL,
    Publisher VARCHAR(100) NOT NULL,
    Pages INT(11) NOT NULL,
    Format TEXT DEFAULT '000х000/00',
    Date DATE DEFAULT NULL,
    Circulation INT(11) NOT NULL,
    Topic VARCHAR(100) NOT NULL,
    Category VARCHAR(100) NOT NULL,
    PRIMARY KEY (`№`),
    INDEX Code (Code)
);

-- d. Створення додаткового індексу
CREATE INDEX idx_Publisher ON table_1(Publisher);

3. SQL-script:

-- a. Додавання в таблицю нового текстового поля "Автор", розміром 15 символів
ALTER TABLE table_1
ADD COLUMN Author VARCHAR(15);

-- b. Збільшення розміру текстового поля "Автор" до 20 символів
ALTER TABLE table_1
MODIFY COLUMN Author VARCHAR(20);

-- c. Видалення текстового поля "Автор" з таблиці
ALTER TABLE table_1
DROP COLUMN Author;

-- d. Зміна порядку сортування індексу і призначення його унікальності
ALTER TABLE table_1
DROP INDEX idx_Publisher,
ADD UNIQUE INDEX idx_Publisher UNIQUE (Publisher DESC);

-- e. Видалення індексу
DROP INDEX idx_Publisher ON table_1;
